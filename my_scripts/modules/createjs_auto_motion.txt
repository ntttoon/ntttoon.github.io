var ac = this.children;

function initPropsMcs(){
	for(var i=0;i<ac.length;i++){
		ac[i].alpha = 0;
		ac[i].scaleX = 2;
		ac[i].scaleY = 2;
		//ac[i].x = ac[i].x - 200;
		//ac[i].rotation = 180;
	}
}

stage.on("drawstart", initPropsMcs, this, true);
